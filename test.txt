  body {
    background: none;
  }
  .page {
    margin: 0;
    box-shadow: none;
    page-break-after: always;
  }
}




import { Component, ViewChild } from '@angular/core';
import { EditorComponent } from '@tinymce/tinymce-angular';

@Component({
  selector: 'app-editor',
  templateUrl: './editor.component.html',
  styleUrls: ['./editor.component.css']
})
export class EditorComponent {

  @ViewChild(EditorComponent) editor!: EditorComponent;

  editorInit = {
    height: '100%',
    menubar: false,
    plugins: 'lists table',
    toolbar: 'undo redo | bold italic | alignleft aligncenter alignright | bullist numlist',
    content_style: `
      body { margin: 0; background: #e5e5e5; }
    `,
    setup: (editor: any) => {
      editor.on('input keyup paste', () => {
        setTimeout(() => this.handleAutoPageBreak(editor), 0);
      });
    }
  };

  ngAfterViewInit() {
    setTimeout(() => this.initializeFirstPage(), 500);
  }

  initializeFirstPage() {
    const editor = this.editor.editor;
    if (!editor.getContent()) {
      editor.setContent(`<div class="page"><p></p></div>`);
    }
  }

  handleAutoPageBreak(editor: any) {
    const body = editor.getBody();
    const pages = body.querySelectorAll('.page');

    pages.forEach((page: HTMLElement) => {
      const maxHeight = page.clientHeight;
      const content = Array.from(page.children);

      let currentHeight = 0;

      content.forEach((el: any) => {
        currentHeight += el.offsetHeight;

        if (currentHeight > maxHeight - 10) {
          let newPage = document.createElement('div');
          newPage.className = 'page';

          let nextSibling = el.nextSibling;
          newPage.appendChild(el);

          while (nextSibling) {
            const temp = nextSibling.nextSibling;
            newPage.appendChild(nextSibling);
            nextSibling = temp;
          }

          page.after(newPage);
        }
      });
    });
  }
}



<div class="editor-wrapper">
