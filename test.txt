ol.multilevel,
ol.multilevel ol {
  counter-reset: item;
  list-style: none;
  margin: 0 0 0 1.5rem;
  padding: 0;
}

ol.multilevel li {
  counter-increment: item;
  position: relative;
  margin: 0.25rem 0;
  padding-left: 2.5rem;
}

ol.multilevel li::before {
  content: counters(item, ".") " ";
  position: absolute;
  left: 0;
  top: 0;
}

setup: (editor) => {
  editor.on('NodeChange', () => {
    const body = editor.getBody();
    body.querySelectorAll('ol').forEach((ol) => {
      ol.classList.add('multilevel');
    });
  });
}

setup: (editor) => {
  editor.ui.registry.addButton('customNumList', {
    text: 'Legal List',
    onAction: () => {
      editor.execCommand('InsertOrderedList');
      const node = editor.selection.getNode();
      const ol = node.closest('ol');
      if (ol) {
        ol.classList.add('multilevel');
      }
    }
  });
}

this.editorInit = {
  promotion: false,
  branding: false,
  menubar: 'file edit view insert format tools table help',
  plugins: 'lists advlist code',
  toolbar: 'undo redo | numlist bullist | outdent indent | code',
  content_css: ['/assets/editor-list.css'],
  extended_valid_elements: 'ol[class],li',
  setup: (editor) => {
    const applyMultiLevelClass = () => {
      const body = editor.getBody();
      if (!body) return;

      body.querySelectorAll('ol').forEach((ol) => {
        ol.classList.add('multilevel');
      });
    };

    editor.on('init', applyMultiLevelClass);
    editor.on('SetContent', applyMultiLevelClass);
    editor.on('Change', applyMultiLevelClass);
    editor.on('NodeChange', applyMultiLevelClass);
    editor.on('ExecCommand', () => setTimeout(applyMultiLevelClass, 0));
  }
};
