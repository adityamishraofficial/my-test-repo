private prepareNestedOrderedLists(root: HTMLElement): void {

  orderedLists.forEach((list) => {
    const ol = list as HTMLOListElement;
    ol.classList.add('legal-ol');
    ol.style.margin = '0 0 1rem 0';
    ol.style.paddingLeft = '2rem';

  const items = root.querySelectorAll('li');
  items.forEach((li) => {

    item.style.margin = '0.35rem 0';
  });
}
