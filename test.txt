export function transformDate(input: string) {
  // Expected input: "DD/MM/YYYY"
  const [day, month, year] = input.split('/');

  if (!day || !month || !year) {
    throw new Error("Invalid date format. Expected DD/MM/YYYY");
  }

  // Create Date object
  const date = new Date(Number(year), Number(month) - 1, Number(day));

  // Helper to format day suffix in English (st, nd, rd, th)
  const getDaySuffix = (d: number): string => {
    if (d > 3 && d < 21) return `${d}th`;
    switch (d % 10) {
      case 1: return `${d}st`;
      case 2: return `${d}nd`;
      case 3: return `${d}rd`;
      default: return `${d}th`;
    }
  };

  // Format: In words EN → "16th June 2025"
  const english = `${getDaySuffix(Number(day))} ${date.toLocaleString("en-GB", { month: "long" })} ${year}`;

  // Format: In words FR → "16 juin 2025"
  const french = date.toLocaleDateString("fr-FR", { day: "numeric", month: "long", year: "numeric" });

  // Format: American standard → "MM/DD/YYYY"
  const american = `${month}/${day}/${year}`;

  return {
    english,
    french,
    american
  };
}
