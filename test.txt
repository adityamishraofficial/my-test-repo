type CurrencyInfo = {
  english: string;
  french: string;
  symbol: string;
};

function transposeCurrency(data: string[][]): Record<string, CurrencyInfo> {
  const result: Record<string, CurrencyInfo> = {};

  data.forEach(([code, english, french, symbol]) => {
    result[code] = {
      english,
      french,
      symbol,
    };
  });

  return result;
}
