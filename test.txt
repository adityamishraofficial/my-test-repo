this.oauthService.refreshToken().then(resp => {
  console.log("Refresh response:", resp);

  if (resp && resp['access_token']) {
    this.oauthService.processIdToken(resp['id_token'], resp['access_token'])
      .then(result => {
        this.oauthService.processAccessTokenResponse(
          resp,               // the full JSON
          result.idToken,     // parsed ID token
          true                // save tokens
        );
        console.log("âœ… Tokens updated in storage");
      });
  }
});
