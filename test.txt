getPlainTextPreview(html: string): string {
  if (isPlatformBrowser(this.platformId)) {
    const parser = new DOMParser();
    const doc = parser.parseFromString(html, 'text/html');
    const plainText = doc.body.textContent || '';
    return plainText.substring(0, 20) + (plainText.length > 20 ? '...' : '');
  }
  return '';
}


let previewContent = editorContent.replace(/{{([^}]+)}}/g, (match: any, key: string) => {
  const match2 = this.variables.find(
    (item) =>
      item.variableName + '_' + item.groupName === key.trim() ||
      item.variableId === key.trim()
  );
  return match2?.variableValue ?? match;
});
