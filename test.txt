private buildPreviewHtml(editorHtml: string): string {
  const parser = new DOMParser();
  const doc = parser.parseFromString(`<div id="root">${editorHtml}</div>`, 'text/html');
  const root = doc.getElementById('root');

  if (!root) {
    return editorHtml;
  }

  // Step 1: fix malformed nested list structure
  this.normalizeNestedOrderedLists(root);

  // Step 2: apply class and cleanup attributes
  const orderedLists = root.querySelectorAll('ol');
  orderedLists.forEach((ol) => {
    ol.classList.add('hierarchical-list');
    ol.removeAttribute('type');
    ol.removeAttribute('start');
    ol.style.listStyle = 'none';
  });

  // Step 3: cleanup li inline list styling if any
  const listItems = root.querySelectorAll('li');
  listItems.forEach((li) => {
    (li as HTMLElement).style.listStyle = 'none';
  });

  return `
    <!DOCTYPE html>
    <html>
      <head>
        <meta charset="UTF-8" />
        ${this.getHierarchicalListStyle()}
      </head>
      <body>
        ${root.innerHTML}
      </body>
    </html>
  `;
}

Add this new method

This is the missing piece in your logic:

private normalizeNestedOrderedLists(root: HTMLElement): void {
  const allOrderedLists = Array.from(root.querySelectorAll('ol'));

  allOrderedLists.forEach((ol) => {
    let node = ol.firstChild;

    while (node) {
      const nextNode = node.nextSibling;

      if (
        node.nodeType === Node.ELEMENT_NODE &&
        (node as HTMLElement).tagName.toLowerCase() === 'ol'
      ) {
        let previousElement = node.previousSibling;

        while (previousElement && previousElement.nodeType !== Node.ELEMENT_NODE) {
          previousElement = previousElement.previousSibling;
        }

        if (
          previousElement &&
          (previousElement as HTMLElement).tagName.toLowerCase() === 'li'
        ) {
          previousElement.appendChild(node);
        }
      }

      node = nextNode;
    }
  });
}

this is invalid.

It must become:

  <li>fdsfsdfd</li>
  <li>
    fdsfsdds
    <ol>
      <li>fds</li>
    </ol>
  </li>
</ol>

That normalization method fixes this kind of issue.

Fix 2: your CSS should be slightly improved

Use this version instead of the current one:

private getHierarchicalListStyle(): string {
  return `
    <style>
      .hierarchical-list,
      .hierarchical-list ol {
        list-style: none !important;
        margin: 0 !important;
        padding-left: 42px !important;
        counter-reset: item;
      }

      .hierarchical-list li {
        display: block;
        position: relative;
        margin: 6px 0;
      }

      .hierarchical-list li::before {
        content: counters(item, ".") ". ";
        counter-increment: item;
        position: absolute;
        left: -42px;
        width: 36px;
        text-align: right;
        white-space: nowrap;
      }
    </style>
  `;
}

Why this is better
	•	!important helps when preview/PDF renderer injects default list styling
	•	white-space: nowrap keeps numbering together
	•	aligned offsets improve indentation
