private normalizeEditorContent(content: string): string {
  const container = document.createElement('div');
  container.innerHTML = content;

  // Important:
  // Only remove rendered preview wrappers if they exist.
  // Do NOT rebuild ol/li from rendered divs.
  this.unwrapRenderedPreview(container);

  return container.innerHTML;
}

private unwrapRenderedPreview(root: HTMLElement): void {
  const renderedPreview = root.querySelector('[data-rendered-preview="true"]');

  if (!renderedPreview) {
    return;
  }

  // If original semantic HTML is stored, restore it
  const originalHtml = renderedPreview.getAttribute('data-original-html');

  if (originalHtml) {
    renderedPreview.replaceWith(...this.htmlToNodes(originalHtml));
    return;
  }

  // fallback: remove only wrapper, keep children
  renderedPreview.replaceWith(...Array.from(renderedPreview.childNodes));
}

private htmlToNodes(html: string): Node[] {
  const template = document.createElement('template');
  template.innerHTML = html;
  return Array.from(template.content.childNodes);
}
