ol.multilevel,
ol.multilevel ol {
  counter-reset: item;
  list-style: none;
  margin: 0 0 0 1.5rem;
  padding: 0;
}

ol.multilevel li {
  counter-increment: item;
  position: relative;
  margin: 0.25rem 0;
  padding-left: 2.5rem;
}

ol.multilevel li::before {
  content: counters(item, ".") " ";
  position: absolute;
  left: 0;
  top: 0;
}

setup: (editor) => {
  editor.on('NodeChange', () => {
    const body = editor.getBody();
    body.querySelectorAll('ol').forEach((ol) => {
      ol.classList.add('multilevel');
    });
  });
}

setup: (editor) => {
  editor.ui.registry.addButton('customNumList', {
    text: 'Legal List',
    onAction: () => {
      editor.execCommand('InsertOrderedList');
      const node = editor.selection.getNode();
      const ol = node.closest('ol');
      if (ol) {
        ol.classList.add('multilevel');
      }
    }
  });
}
